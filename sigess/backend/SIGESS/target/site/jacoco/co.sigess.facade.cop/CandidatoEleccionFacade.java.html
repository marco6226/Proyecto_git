package co.sigess.facade.cop;

import co.sigess.entities.cop.CandidatoEleccion;
import co.sigess.facade.com.AbstractFacade;
import java.util.List;
import javax.ejb.Stateless;
import javax.persistence.EntityManager;
import javax.persistence.PersistenceContext;
import javax.persistence.Query;

/**
 *
 * @author fmoreno
 */
@Stateless
public class CandidatoEleccionFacade extends AbstractFacade<CandidatoEleccion> {

    @PersistenceContext(unitName = "SIGESS_PU")
    private EntityManager em;

    @Override
    protected EntityManager getEntityManager() {
        return em;
    }

    public CandidatoEleccionFacade() {
        super(CandidatoEleccion.class);
    }
    
    public List<CandidatoEleccion> buscarCandidatos(Long eleccId) {
        Query q = this.em.createQuery("SELECT c FROM CandidatoEleccion c WHERE c.eleccionCopasst.id = ?1");
        q.setParameter(1, eleccId);
        List<CandidatoEleccion> list = q.getResultList();
        return list;
    }
}
