package co.sigess.facade.cop;

import co.sigess.entities.com.Mensaje;
import co.sigess.entities.com.TipoMensaje;
import co.sigess.entities.cop.CandidatoEleccion;
import co.sigess.entities.cop.EleccionCopasst;
import co.sigess.entities.cop.VotacionEleccion;
import co.sigess.exceptions.UserMessageException;
import co.sigess.facade.com.AbstractFacade;
import java.util.Date;
import java.util.List;
import javax.ejb.EJB;
import javax.ejb.Stateless;
import javax.persistence.EntityManager;
import javax.persistence.PersistenceContext;
import javax.persistence.Query;

/**
 *
 * @author fmoreno
 */
@Stateless
public class EleccionCopasstFacade extends AbstractFacade<EleccionCopasst> {

    @EJB
    private CandidatoEleccionFacade candidatoEleccionFacade;

    @EJB
    private VotacionEleccionFacade votacionEleccionFacade;

    @PersistenceContext(unitName = "SIGESS_PU")
    private EntityManager em;

    @Override
    protected EntityManager getEntityManager() {
        return em;
    }

    public EleccionCopasstFacade() {
        super(EleccionCopasst.class);
    }

    public List<CandidatoEleccion> buscarCandidatos(Long eleccId) {
        Query q = this.em.createQuery("SELECT c FROM CandidatoEleccion c WHERE c.eleccionCopasst.id = ?1");
        q.setParameter(1, eleccId);
        List<CandidatoEleccion> list = q.getResultList();
        return list;
    }

    public CandidatoEleccion adicionarCandidato(CandidatoEleccion candidato) throws Exception {
        return this.candidatoEleccionFacade.create(candidato);
    }

    public VotacionEleccion votar(VotacionEleccion voto) throws Exception {
        if (voto.getCandidatoEleccionId() == null) {
            throw new UserMessageException("CANDIDATO NO ESPECIFICADO", "No se ha especificado el candidato votado", TipoMensaje.warn);
        }
        if (voto.getEleccionCopasstId() == null) {
            throw new UserMessageException("CONVOCATORIA NO ESPECIFICADA", "No se ha especificado la convocatoria en que est√° votando", TipoMensaje.warn);
        }
        if (voto.getUsuarioId() == null) {
            throw new UserMessageException("USUARIO NO ESPECIFICADO", "No se ha especificado el usuario votante", TipoMensaje.warn);
        }
        Query q = this.em.createQuery("SELECT v FROM VotacionEleccion v WHERE v.usuarioId = ?1 AND v.eleccionCopasstId = ?2");
        q.setParameter(1, voto.getUsuarioId());
        q.setParameter(2, voto.getEleccionCopasstId());
        List<VotacionEleccion> list = q.getResultList();
        if (list != null && !list.isEmpty()) {
            Mensaje msg = new Mensaje("VOTACION YA REALIZADA", "Su voto ya habia sido registrado anteriormente.", TipoMensaje.warn, Mensaje.COD_VOTO_PREVIO_REGISTRADO);
            throw new UserMessageException(msg);
        }
        voto.setFechaHora(new Date());
        return this.votacionEleccionFacade.create(voto);
    }

}
