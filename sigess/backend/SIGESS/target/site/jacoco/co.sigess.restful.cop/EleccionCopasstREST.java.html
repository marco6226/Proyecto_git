package co.sigess.restful.cop;

import co.sigess.entities.cop.CandidatoEleccion;
import co.sigess.entities.cop.EleccionCopasst;
import co.sigess.entities.cop.VotacionEleccion;
import co.sigess.facade.cop.EleccionCopasstFacade;
import co.sigess.restful.FilterQuery;
import co.sigess.restful.ServiceREST;
import co.sigess.restful.security.Secured;
import co.sigess.util.Util;
import java.util.List;
import javax.ws.rs.DELETE;
import javax.ws.rs.GET;
import javax.ws.rs.POST;
import javax.ws.rs.PUT;
import javax.ws.rs.Path;
import javax.ws.rs.PathParam;
import javax.ws.rs.Produces;
import javax.ws.rs.core.MediaType;
import javax.ws.rs.core.Response;

/**
 *
 * @author fmoreno
 */
@Secured
@Path("eleccion")
public class EleccionCopasstREST extends ServiceREST {
    
    public EleccionCopasstREST() {
        super(EleccionCopasstFacade.class, "empresaId");
    }
    
    @Override
    public Response findWithFilter(FilterQuery filterQuery) {
        return super.findWithFilter(filterQuery); //To change body of generated methods, choose Tools | Templates.
    }
    
    @POST
    @Produces({MediaType.APPLICATION_XML, MediaType.APPLICATION_JSON})
    public Response create(EleccionCopasst eleccion) {
        try {
            eleccion.setEmpresaId(super.getEmpresaIdRequestContext());
            eleccion = ((EleccionCopasstFacade) super.beanInstance).create(eleccion);
            return Response.ok(eleccion).build();
        } catch (Exception ex) {
            return Util.manageException(ex, EleccionCopasstREST.class);
        }
    }
    
    @PUT
    @Produces({MediaType.APPLICATION_XML, MediaType.APPLICATION_JSON})
    public Response edit(EleccionCopasst eleccion) {
        try {
            eleccion = ((EleccionCopasstFacade) super.beanInstance).edit(eleccion);
            return Response.ok(eleccion).build();
        } catch (Exception ex) {
            return Util.manageException(ex, EleccionCopasstREST.class);
        }
    }
    
    @DELETE
    @Path("{eleccionCopasstId}")
    @Produces({MediaType.APPLICATION_XML, MediaType.APPLICATION_JSON})
    public Response delete(@PathParam("actaId") Long actaId) {
        try {
            EleccionCopasst elec = new EleccionCopasst(actaId);
            ((EleccionCopasstFacade) super.beanInstance).remove(elec);
            return Response.ok(elec).build();
        } catch (Exception ex) {
            return Util.manageException(ex, EleccionCopasstREST.class);
        }
    }
    
    
    @GET
    @Path("{eleccionId}/candidatos")
    @Produces({MediaType.APPLICATION_XML, MediaType.APPLICATION_JSON})
    public Response findCandidatos(@PathParam("eleccionId") Long eleccId) {
        try {
            List<CandidatoEleccion> list = ((EleccionCopasstFacade) super.beanInstance).buscarCandidatos(eleccId);
            return Response.ok(list).build();
        } catch (Exception ex) {
            return Util.manageException(ex, EleccionCopasstREST.class);
        }
    }
    
    @POST
    @Path("{eleccionId}/candidatos")
    @Produces({MediaType.APPLICATION_XML, MediaType.APPLICATION_JSON})
    public Response adicionarCandidato(@PathParam("eleccionId") Long eleccId, CandidatoEleccion candidato) {
        try {
            candidato.setEleccionCopasst(new EleccionCopasst(eleccId));
            candidato = ((EleccionCopasstFacade) super.beanInstance).adicionarCandidato(candidato);
            return Response.ok(candidato).build();
        } catch (Exception ex) {
            return Util.manageException(ex, EleccionCopasstREST.class);
        }
    }
         
    @POST
    @Path("voto")
    @Produces({MediaType.APPLICATION_XML, MediaType.APPLICATION_JSON})
    public Response votar(VotacionEleccion voto) {
        try {
            voto.setUsuarioId(super.getUsuarioRequestContext().getId());
            voto = ((EleccionCopasstFacade) super.beanInstance).votar(voto);
            return Response.ok(voto).build();
        } catch (Exception ex) {
            return Util.manageException(ex, EleccionCopasstREST.class);
        }
    }
}
